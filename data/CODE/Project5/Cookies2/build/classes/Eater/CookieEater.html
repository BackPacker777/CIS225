19     public CookieEater(String fileName) throws FileNotFoundException {
20         plate = new ArrayList<>();
21         fileReader = new File(fileName);
22         streamReader = new Scanner(fileReader);
23         readData();
24         streamReader.close();

28     private void readData() {
29         String[] data;
30         while (streamReader.hasNextLine()) {
31             String line = streamReader.nextLine();
32             data = line.trim().split(",");
33             populatePlate(data);
34         }
35     }
36     
37     private void populatePlate(String[] data) {
38         final int PEANUTBUTTER = 0;
39         final int OATMEALRAISIN = 1;
40         final int CHOCOLATECHIP = 2;
41         int cookieType = Integer.parseInt(data[0]);
42         int cookieQty = Integer.parseInt(data[1]);
43         for (int i = 0; i < cookieQty; i++) {
44             switch (cookieType) {
45                 case PEANUTBUTTER:
46                     plate.add(cookie = new PeanutButterCookie());
47                     break;
48                 case OATMEALRAISIN:
49                     plate.add(cookie = new OatmealRaisinCookie());
50                     break;
51                 case CHOCOLATECHIP:
52                     plate.add(cookie = new ChocolateChipCookie());
53                     break;
54                 default:
55                     System.out.println("INVALID COOKIE!");
56                     System.exit(1);
57             }
58         }
59     }
60     
61     private void printCookies() {
62         for (int i = 0; i < plate.size(); i++) {
63             System.out.println(plate.get(i));
64         }
65     }
